		%		}{%
		\clearpage%
		%		}%
		\phantomsection% damit die Seitenzahl auf jeden Fall stimmt
		\nopagebreak%
		\addcontentsline{toc}{chapter}{\texorpdfstring{#1}{#2}}% 
		\nopagebreak%
	}{%
		\pdfbookmark[0]{#2}{#3}% hier passte die Seitenzahl bisher immer, daher keine \phantomsection nötig
	}%
}
